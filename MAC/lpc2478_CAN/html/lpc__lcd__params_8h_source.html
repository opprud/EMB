00056                                              clocks */
00057   unsigned char           v_front_porch;        /* Vertical front porch in

00060                                              clocks (minimum 1 clock) */
00061   unsigned short          lines_per_panel;      /* Lines per panel (vertical
00062                                              resolution) */
00063   unsigned char           invert_output_enable; /* Invert output enable, 1 =
00064                                              invert*/

00069   unsigned char           ac_bias_frequency;    /* AC bias frequency in
00070                                              clocks (minimum 1) */
00071   unsigned char           bits_per_pixel;       /* Maximum bits per pixel the
00072                                              display supports */

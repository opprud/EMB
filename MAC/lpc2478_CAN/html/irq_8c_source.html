00058       
00059     VICIntEnClr = 1 << IntNumber;       /* Disable Interrupt */
00060     if ( IntNumber >= VIC_SIZE )
00061     {
00062                 return ( FALSE );

00065     {
00066                 /* find first un-assigned VIC address for the handler */
00067                 vect_addr = (int *)(VIC_BASE_ADDR + VECT_ADDR_INDEX + IntNumber*4);
00068                 vect_cntl = (int *)(VIC_BASE_ADDR + VECT_CNTL_INDEX + IntNumber*4);
00069                 *vect_addr = (int)HandlerAddr;  /* set interrupt vector */
00070                 *vect_cntl = Priority;

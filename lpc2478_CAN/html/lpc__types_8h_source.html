00095 /* _BIT(n) sets the bit at position "n"
00096  * _BIT(n) is intended to be used in "OR" and "AND" expressions:
00097  * e.g., "(_BIT(3) | _BIT(7))".
00098  */
00099 #undef _BIT
00100 /* Set bit macro */
00101 #define _BIT(n) (((UNS_32)(1)) << (n))
00102 
00103 /* _SBF(f,v) sets the bit field starting at position "f" to value "v".
00104  * _SBF(f,v) is intended to be used in "OR" and "AND" expressions:
00105  * e.g., "((_SBF(5,7) | _SBF(12,0xF)) & 0xFFFF)"
00106  */
00107 #undef _SBF
00108 /* Set bit field macro */
00109 #define _SBF(f,v) (((UNS_32)(v)) << (f))
00110 
00111 /* _BITMASK constructs a symbol with 'field_width' least significant
00112  * bits set.
00113  * e.g., _BITMASK(5) constructs '0x1F', _BITMASK(16) == 0xFFFF
00114  * The symbol is intended to be used to limit the bit field width
00115  * thusly:
00116  * <a_register> = (any_expression) & _BITMASK(x), where 0 < x <= 32.
00117  * If "any_expression" results in a value that is larger than can be
00118  * contained in 'x' bits, the bits above 'x - 1' are masked off.  When
00119  * used with the _SBF example above, the example would be written:
00120  * a_reg = ((_SBF(5,7) | _SBF(12,0xF)) & _BITMASK(16))
00121  * This ensures that the value written to a_reg is no wider than
00122  * 16 bits, and makes the code easier to read and understand.
00123  */
00124 #undef _BITMASK
00125 /* Bitmask creation macro */
00126 #define _BITMASK(field_width) ( _BIT(field_width) - 1)

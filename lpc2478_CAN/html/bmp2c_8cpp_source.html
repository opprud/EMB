00296 // do_bit_manipulation  //
00298 //
00299 //perform bit manipulation on unsigned int value.
00300 //the input value is transformed according to the operations stored in the bit_assignation vector
00301 //each element of the vector describe the value of the corresponding bit 
00302 //(the output value is always right justified)
00303 //if the element is CST_BIT_BASE or CST_BIT_BASE+1, constant bit 0 or 1 respectively will be copied
00304 //if the element is below CST_BIT_BASE, a bit from the input value will be copied
00305 //    the value of the element indicates the index of the bit to copy
00306 //
00307 //example: 
00308 //input value (in binary) = 01000001
00309 //bit_assignation    = c1 c0  0  1   |  6  7 c0 c0 c1  (c1=CST_BIT_BASE+1 and c0=CST_BIT_BASE)              
00310 //output (in binary) =  1  0  1  0   |  1  0  0  0  1
00311 void do_bit_manipulation(unsigned int input, unsigned int &output, vector<int> bit_assignation)
00312 try
00313 {
00314    for(unsigned int bit_cnt=0;bit_cnt<bit_assignation.size();bit_cnt++)
00315    {
00316       unsigned int dest_bit=bit_assignation.size()-bit_cnt-1;
00317       bool bit;
00318       //compute bit value
00319       if(bit_assignation[bit_cnt]>=CST_BIT_BASE)//this bit is stuck to a constant value
00320       {
00321          if(bit_assignation[bit_cnt]>CST_BIT_BASE+1)
00322             throw error_msg("internal error: bit_assignation[bit_cnt]>CST_BIT_BASE+1");
00323          bit=bit_assignation[bit_cnt]-CST_BIT_BASE;
00324       }
00325       else
00326       {
00327          if(bit_assignation[bit_cnt]>=0)
00328             bit=(input>>bit_assignation[bit_cnt])&0x01;//get the pointed bit
00329          else
00330             bit=!((input>>(-bit_assignation[bit_cnt]-1))&0x01);//get the complement of the pointed bit
00331       }
00332       //modify output
00333       if(bit==0)
00334          output&=~(((unsigned int)bit)<<dest_bit);
00335       else
00336          output|=(((unsigned int)bit)<<dest_bit);
00337    }
00338 }
